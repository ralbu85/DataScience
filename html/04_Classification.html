def accuracy(y_hat, y):
    hit=0
    for i in range(len(y)):
        if y[i]==y_hat[i]:
            hit += 1
    return hit/len(y)

def predict_th(logreg, th=0.5):
    result=[]
    for i in logreg.predict_proba(X)[:,1]:
        if i>th:
            result.append(1)
        else:
            result.append(0)
    return np.array(result)

y_hat=predict_th(logreg,0.7)
