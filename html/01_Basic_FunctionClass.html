def sign(x):
    if x>0:
        return 'positive'
    elif x<0:
        return 'negative'
    else:
        return 'zero'

In [26]:
def hello(name, loud=False):
    if loud:
        print('HELLO ', name)
    else:
        print('Hello ', name)
        
hello('Bob') # Prints "Hello, Bob"
hello('Fred', loud=True)  # Prints "HELLO, FRED!"

Hello  Bob
HELLO  Fred

class Calculator:
    # 생성자
    def __init__(self,num):
        self.x=num  
    def add(self,y):
        self.x += y
    def sub(self,y):
        self.x -= y
    def get(self):
        return self.x

cal = Calculator(20)
cal.add(2)
print(cal.get())
