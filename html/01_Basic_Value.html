5. 반복을 활용한 리스트 값 복사¶

for x in nums:
    squares.append(x**2)
print(squares)

In [30]:
nums = [0,1,2,3,4]
squares = [x**2 for x in nums]
print(squares)

[0, 1, 4, 9, 16]
In [31]:
nums = [0, 1, 2, 3, 4]
even_squares = [x ** 2 for x in nums if x % 2 == 0]
print(even_squares)  

[0, 4, 16]
